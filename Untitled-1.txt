    });

    // fallback: se o áudio ainda não tocou, inicia na primeira interação do usuário
    (function () {
      const bgAudio = document.getElementById('bgAudio');
      let attempted = false;
      function tryPlay() {
        if (attempted || !bgAudio) return;
        attempted = true;
        bgAudio.play().catch(() => { });
        document.removeEventListener('click', tryPlay);
        document.removeEventListener('keydown', tryPlay);
        document.removeEventListener('touchstart', tryPlay);
      }
      document.addEventListener('click', tryPlay);
      document.addEventListener('keydown', tryPlay);
      document.addEventListener('touchstart', tryPlay);
    })();

    // Galeria toggle
    const galleryToggle = document.getElementById('galleryToggle');
    const galleryBody = document.getElementById('galleryBody');
    galleryToggle?.addEventListener('click', () => {
      const isOpen = galleryBody.classList.toggle('open');
      galleryToggle.innerHTML = `<i class="fa-regular fa-images mr-2" style="color: var(--gold);"></i>${isOpen ? 'Fechar' : 'Abrir'}`;
      if (isOpen) gsap.fromTo("#galleryBody", { opacity: 0, y: 6 }, { opacity: 1, y: 0, duration: .35, ease: "power2.out" });
    });

    // Lightbox
    const thumbs = Array.from(document.querySelectorAll('.thumb'));
    const lightbox = document.getElementById('lightbox');
    const lightboxImg = document.getElementById('lightboxImg');
    const lightboxClose = document.getElementById('lightboxClose');
    const lightboxPrev = document.getElementById('lightboxPrev');
    const lightboxNext = document.getElementById('lightboxNext');
    const lightboxLabel = document.getElementById('lightboxLabel');
    let currentIndex = 0;

    function openLightbox(index) {
      currentIndex = index;
      const item = thumbs[currentIndex];
      const src = item.getAttribute('data-full') || item.querySelector('img')?.src;
      const label = item.getAttribute('data-label') || 'Galeria';
      if (lightboxImg) lightboxImg.src = src;
      if (lightboxLabel) lightboxLabel.textContent = label;
      lightbox?.classList.add('open');
      lightbox?.setAttribute('aria-hidden', 'false');
      gsap.fromTo(".lightbox-shell", { opacity: 0, y: 14, scale: 0.985 }, { opacity: 1, y: 0, scale: 1, duration: 0.6, ease: "power3.out" });
    }

    function closeLightbox() {
      if (!lightbox?.classList.contains('open')) return;
      gsap.to(".lightbox-shell", {
        opacity: 0, y: 10, scale: 0.99, duration: 0.35, ease: "power2.in",
        onComplete: () => {
          lightbox.classList.remove('open');
          lightbox.setAttribute('aria-hidden', 'true');
          if (lightboxImg) lightboxImg.src = '';
        }
      });
    }

    function nav(delta) {
      currentIndex = (currentIndex + delta + thumbs.length) % thumbs.length;
      const item = thumbs[currentIndex];
      const src = item.getAttribute('data-full') || item.querySelector('img')?.src;
      const label = item.getAttribute('data-label') || 'Galeria';
      if (lightboxImg) lightboxImg.src = src;
      if (lightboxLabel) lightboxLabel.textContent = label;
      gsap.fromTo("#lightboxImg", { opacity: 0, scale: 0.995 }, { opacity: 1, scale: 1, duration: 0.35, ease: "power2.out" });
    }

    thumbs.forEach((t, i) => t.addEventListener('click', () => openLightbox(i)));
    lightboxClose?.addEventListener('click', closeLightbox);

    // Evento para abrir o overlay de comentários a partir do lightbox atual
    document.getElementById('lightboxCommentBtn')?.addEventListener('click', () => {
      openPhotoHover(currentIndex);
    });
    lightboxPrev?.addEventListener('click', () => nav(-1));
    lightboxNext?.addEventListener('click', () => nav(1));
    lightbox?.addEventListener('click', (e) => { if (e.target === lightbox) closeLightbox(); });

    window.addEventListener('keydown', (e) => {
      if (!lightbox?.classList.contains('open')) return;
      if (e.key === 'Escape') closeLightbox();
      if (e.key === 'ArrowLeft') nav(-1);
      if (e.key === 'ArrowRight') nav(1);
    });

    // ===========================
    // Photo Hover Expand Modal
    // ===========================
    const photoHoverOverlay = document.getElementById('photoHoverOverlay');
    const photoHoverImg = document.getElementById('photoHoverImg');
    const photoHoverLabel = document.getElementById('photoHoverLabel');
    const photoHoverNote = document.getElementById('photoHoverNote');
    const photoHoverClose = document.getElementById('photoHoverClose');
    const photoHoverSave = document.getElementById('photoHoverSave');

    function savePhotoNote(photoIndex) {
      const notes = JSON.parse(localStorage.getItem('photoNotes') || '{}');
      const photoPath = thumbs[photoIndex]?.getAttribute('data-full') || '';
      notes[photoPath] = photoHoverNote.value;
      localStorage.setItem('photoNotes', JSON.stringify(notes));
    }

    function loadPhotoNote(photoIndex) {
      const notes = JSON.parse(localStorage.getItem('photoNotes') || '{}');
      const photoPath = thumbs[photoIndex]?.getAttribute('data-full') || '';
      photoHoverNote.value = notes[photoPath] || '';
    }

    function openPhotoHover(index) {
      const item = thumbs[index];
      const src = item.getAttribute('data-full') || item.querySelector('img')?.src;
      const label = item.getAttribute('data-label') || 'Galeria';

      if (photoHoverImg) photoHoverImg.src = src;
      if (photoHoverLabel) photoHoverLabel.textContent = label;

      loadPhotoNote(index);

      photoHoverOverlay?.classList.add('active');
      gsap.fromTo(".photo-expanded-shell", { opacity: 0, y: 20, scale: 0.9 }, { opacity: 1, y: 0, scale: 1, duration: 0.4, ease: "power3.out" });

      // Focus no textarea automaticamente
      photoHoverNote?.focus();

      // Guardar o índice da foto atual para salvar depois
      photoHoverOverlay._currentIndex = index;
    }

    function closePhotoHover() {
      if (!photoHoverOverlay?.classList.contains('active')) return;
      gsap.to(".photo-expanded-shell", {
        opacity: 0, y: 14, scale: 0.95, duration: 0.35, ease: "power2.in",
        onComplete: () => {
          photoHoverOverlay.classList.remove('active');
          if (photoHoverImg) photoHoverImg.src = '';
        }
      });
    }

    // Event listeners para foto hover
    // O problema de "flicker" e fotos piscando foi resolvido
    // removendo o evento 'mouseenter' / 'mouseleave' que abria o modal fullscreen,
    // o qual causava imediatamente um 'mouseleave' na foto, fechando o modal
    // em um loop infinito que fazia a tela piscar repetidamente e descontroladamente.

    // Event listeners para os botões do modal hover
    photoHoverClose?.addEventListener('click', closePhotoHover);
    photoHoverSave?.addEventListener('click', () => {
      if (photoHoverOverlay._currentIndex !== undefined) {
        savePhotoNote(photoHoverOverlay._currentIndex);

        // Animação de confirmação
        const originalHTML = photoHoverSave.innerHTML;
        photoHoverSave.innerHTML = '<i class="fa-solid fa-check mr-2" style="color:inherit;"></i> Salvo!';
        photoHoverSave.style.borderColor = 'rgba(197,160,125,.9)';
        photoHoverSave.style.color = '#fff';

        setTimeout(() => {
          photoHoverSave.innerHTML = originalHTML;
          photoHoverSave.style.borderColor = '';
          photoHoverSave.style.color = '';
        }, 1500);
      }
    });

    // Fechar quando clicar no overlay
    photoHoverOverlay?.addEventListener('click', (e) => {
      if (e.target === photoHoverOverlay) closePhotoHover();
    });

    // Fechar com ESC
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && photoHoverOverlay?.classList.contains('active')) {
        closePhotoHover();
      }
    });

    // Pausar o hover quando o modal hover está aberto (para não interferir)
    photoHoverNote?.addEventListener('mouseenter', (e) => {
      e.stopPropagation();
    });

    // Painéis
    document.querySelectorAll('.js-toggle').forEach(btn => {
      btn.addEventListener('click', () => {
        const targetSel = btn.getAttribute('data-target');
        const bodyEl = document.querySelector(targetSel);
        if (!bodyEl) return;
        const isOpen = bodyEl.classList.toggle('open');
        const icon = btn.querySelector('i')?.outerHTML || '';
        btn.innerHTML = `${icon} ${isOpen ? 'Fechar' : 'Abrir'}`;
        if (isOpen) gsap.fromTo(bodyEl, { opacity: 0, y: 6 }, { opacity: 1, y: 0, duration: .35, ease: "power2.out" });
      });
    });

    // Expandir tudo
    const openAllBtn = document.getElementById('openAll');
    openAllBtn?.addEventListener('click', () => {
      if (galleryBody && !galleryBody.classList.contains('open')) {
        galleryBody.classList.add('open');
        galleryToggle.innerHTML = `<i class="fa-regular fa-images mr-2" style="color: var(--gold);"></i> Fechar`;
      }
      document.querySelectorAll('.lux-body').forEach(body => body.classList.add('open'));
      document.querySelectorAll('.js-toggle').forEach(btn => {
        const i = btn.querySelector('i')?.outerHTML || '';
        btn.innerHTML = `${i} Fechar`;
      });
      gsap.fromTo("#app", { opacity: 0.9, y: 8 }, { opacity: 1, y: 0, duration: 0.6, ease: "power2.out" });
      document.querySelector('.gallery-wrap')?.scrollIntoView({ behavior: 'smooth', block: 'start' });
    });

    // Copiar link
    const copyBtn = document.getElementById('copyLink');
    copyBtn?.addEventListener('click', async () => {
      try {
        await navigator.clipboard.writeText(window.location.href);
        copyBtn.innerHTML = `<i class="fa-solid fa-check"></i> Copiado`;
        setTimeout(() => copyBtn.innerHTML = `<i class="fa-solid fa-link"></i> Copiar Link`, 1200);
      } catch (e) {
        const temp = document.createElement('input');
        temp.value = window.location.href;
        document.body.appendChild(temp);
        temp.select();
        document.execCommand('copy');
        document.body.removeChild(temp);
        copyBtn.innerHTML = `<i class="fa-solid fa-check"></i> Copiado`;
        setTimeout(() => copyBtn.innerHTML = `<i class="fa-solid fa-link"></i> Copiar Link`, 1200);
      }
    });

    /* ===========================
       ✅ PLAYER: play overlay + play/pause + mute
    =========================== */
    const tourVideo = document.getElementById('tourVideo');
    const overlay = document.getElementById('videoOverlay');
    const btnPlayPause = document.getElementById('btnPlayPause');
    const btnMute = document.getElementById('btnMute');
    const videoMeta = document.getElementById('videoMeta');
    const videoSelector = document.getElementById('videoSelector');

    function setPlayUI(isPlaying) {
      if (!btnPlayPause) return;
      btnPlayPause.innerHTML = isPlaying
        ? `<i class="fa-solid fa-pause"></i> Pause`
        : `<i class="fa-solid fa-play"></i> Play`;
      if (overlay) overlay.classList.toggle('hidden', isPlaying);
    }

    async function playVideo() {
      if (!tourVideo) return;
      try {
        await tourVideo.play();
        setPlayUI(true);
      } catch (err) {
        // se o navegador bloquear, deixa overlay visível
        setPlayUI(false);
      }
    }

    function pauseVideo() {
      if (!tourVideo) return;
      tourVideo.pause();
      setPlayUI(false);
    }

    function switchVideo(src, label, buttonElement) {
      if (!tourVideo) return;
      const wasPlaying = !tourVideo.paused;
      tourVideo.pause();
      tourVideo.src = src;
      tourVideo.load();
      if (videoMeta) videoMeta.textContent = `Tour ${label} • Estância Aroeira`;
      setPlayUI(false);
      // update active state on buttons
      if (videoSelector) {
        videoSelector.querySelectorAll('.video-switch').forEach(b => b.classList.remove('bg-[#c5a07d]', 'text-black'));
      }
      if (buttonElement) {
        buttonElement.classList.add('bg-[#c5a07d]', 'text-black');
      }
      if (wasPlaying) playVideo();
    }

    overlay?.addEventListener('click', playVideo);
    overlay?.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' || e.key === ' ') playVideo();
    });

    btnPlayPause?.addEventListener('click', () => {
      if (!tourVideo) return;
      if (tourVideo.paused) playVideo();
      else pauseVideo();
    });

    btnMute?.addEventListener('click', () => {
      if (!tourVideo) return;
      tourVideo.muted = !tourVideo.muted;
      btnMute.innerHTML = tourVideo.muted
        ? `<i class="fa-solid fa-volume-xmark"></i> Mute`
        : `<i class="fa-solid fa-volume-high"></i> Som`;
    });

    // handle selector buttons
    videoSelector?.querySelectorAll('.video-switch').forEach(btn => {
      btn.addEventListener('click', () => {
        const src = btn.getAttribute('data-src');
        const label = btn.getAttribute('data-label') || '';
        if (src) switchVideo(src, label, btn);
      });
    });

    // mark first button active on load
    if (videoSelector) {
      const first = videoSelector.querySelector('.video-switch');
      if (first) first.classList.add('bg-[#c5a07d]', 'text-black');
    }

    tourVideo?.addEventListener('ended', () => setPlayUI(false));
    tourVideo?.addEventListener('pause', () => setPlayUI(false));
    tourVideo?.addEventListener('play', () => setPlayUI(true));
  </script>
